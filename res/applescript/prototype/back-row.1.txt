on press_key(key_code)	tell application "System Events"		tell process "Front Row"			key code key_code using {command down}		end tell	end tell	end press_keyon show()	press_key(53) -- ESC 0x35end showon hide()	press_key(122) -- F1 0x7Aend hideon enter()	press_key(36) -- RETURN 0x24end enteron leave()	press_key(53) -- ESC 0x35end leaveon go_left()	press_key(123) -- LEFT 0x7B	end go_lefton go_right()	press_key(124) -- RIGHT 0x7Bend go_righton go_down()	press_key(125) -- DOWN 0x7Dend go_downon go_up()	press_key(126) -- UP 0x7Eend go_up
